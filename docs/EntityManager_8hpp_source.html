150 for (auto& systemPair : systems) {
151 auto& vec = systemPair.second;
152 auto start_junk = std::remove_if(
153 vec.begin(), vec.end(),
154 [typeName](const auto& system) {
155 std::string systemName = typeid(*system).name();
156 std::string strippedName = std::regex_replace(systemName, std::regex("class "), "");
157 return typeName == strippedName;
158 });
159 vec.erase(start_junk, vec.end());
160 }
